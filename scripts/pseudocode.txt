# Get computer choice

FUNCTION getComputerChoice
    INIT choice
    COMPUTE random number 1-3
    SET choice to random number
    CASE choice
        choice == 1
            RETURN "rock"
        choice == 2
            RETURN "paper"
        choice == 3
            RETURN "scissors"
    ENDCASE
ENDFUNCTION

# Get human choice

FUNCTION getHumanChoice

    PRINT "make your choice"
    choice = INPUT string
    choice = LOWERCASE string 

    CASE choice
        choice == "rock"
            RETURN choice
        choice == "paper"
            RETURN choice
        choice == "scissors"
            RETURN choice
        OTHERS
            RETURN "rock" # The assignment did not ask to handle incorrect inputs so I put this here instead
    ENDCASE
ENDFUNCTION



# Function logic

FUNCTION playGame

    INIT humanScore
    SET humanScore = 0
    INIT computerScore
    SET computerScore = 0

    FUNCTION playRound
        INIT computerChoice
        SET computerChoice = getComputerChoice

        INIT humanChoice
        SET humanChoice = getHumanChoice

        CASE
            humanChoice == "rock"
                CASE
                    computerChoice == "rock"
                        RETURN "you tie"
                    computerChoice == "paper"
                        INCREMENT computerScore 1
                        RETURN "you lose"
                    computerChoice == "scissors"
                        INCREMENT humanScore 1
                        RETURN "you win"
                ENDCASE
            humanChoice == "paper"
                CASE
                    computerChoice == "rock"
                        INCREMENT humanScore 1
                        RETURN "you win"
                    computerChoice == "paper"
                        RETURN "you tie"
                    computerChoice == "scissors"
                        INCREMENT computerScore 1
                        RETURN "you lose"
                ENDCASE
            humanChoice == "scissors"
                CASE
                    computerChoice == "rock"
                        INCREMENT computerScore 1
                        RETURN "you lose"
                    computerChoice == "paper"
                        INCREMENT humanScore 1
                        RETURN "you win"
                    computerChoice == "scissors"
                        RETURN "you tie"
                ENDCASE
        ENDCASE
    ENDFUNCTION

    FOR 5 iterations
        playRound

    IF humanScore > computerScore THEN
        PRINT "you win"
    ELSE IF computerScore > humanScore THEN1
        PRINT "you lose"
    ELSE
        PRINT "you tie"
ENDFUNCTION

playGame
